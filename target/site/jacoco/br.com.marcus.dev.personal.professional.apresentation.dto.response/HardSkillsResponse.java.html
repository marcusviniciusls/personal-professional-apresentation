import lombok.NoArgsConstructor;
import lombok.Setter;

import java.util.ArrayList;

@Getter
@Setter

    public void addListFrameworkResponse(FrameworkResponse frameworkResponse){
        this.listFrameworkResponse.add(frameworkResponse);
    }
}
